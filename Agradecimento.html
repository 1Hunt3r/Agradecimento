from flask import Flask, render_template, request, redirect, url_for import sqlite3 app = Flask(__name__) # Conectar ao banco de dados SQLite conn = sqlite3.connect('estoque.db') cursor = conn.cursor() # Criar tabela se n√£o existir cursor.execute(''' CREATE TABLE IF NOT EXISTS mercadorias ( id INTEGER PRIMARY KEY AUTOINCREMENT, nome TEXT, quantidade INTEGER ) ''') conn.commit() @app.route('/') def index(): # Selecionar todas as mercadorias cursor.execute('SELECT * FROM mercadorias') mercadorias = cursor.fetchall() return render_template('index.html', mercadorias=mercadorias) @app.route('/adicionar', methods=['POST']) def adicionar_mercadoria(): # Adicionar mercadoria ao banco de dados nome = request.form['nome'] quantidade = int(request.form['quantidade']) cursor.execute('INSERT INTO mercadorias (nome, quantidade) VALUES (?, ?)', (nome, quantidade)) conn.commit() return redirect(url_for('index')) if __name__ == '__main__': app.run(debug=True)
